#include <Wire.h>
#include <Adafruit_Sensor.h>
#include <Adafruit_BME280.h>

Adafruit_BME280 bme;

void setup() {
  Serial.begin(115200);
  delay(2000);

  Serial.println();
  Serial.println("Iniciando CubeCell + BME280...");

  // Encender VEXT para alimentar el sensor
  pinMode(Vext, OUTPUT);
  digitalWrite(Vext, LOW);   // LOW = encendido en CubeCell

  delay(200); // esperar a que el sensor se estabilice

  // Iniciar I2C
  Wire.begin();

  // Intentar iniciar el BME280 en direccion 0x76
  if (!bme.begin(0x76)) {
    Serial.println("No se encontro el BME280 en 0x76");

    // Probar direccion alternativa 0x77
    if (!bme.begin(0x77)) {
      Serial.println("No se encontro el BME280 en 0x77 tampoco");
      Serial.println("Revisar conexiones y CSB a VEXT/3.3V");
      while (1) {
        delay(1000);
      }
    }
  }

  Serial.println("BME280 detectado correctamente.");
}

void loop() {
  float temperatura = bme.readTemperature();
  float humedad = bme.readHumidity();
  float presion = bme.readPressure() / 100.0F; // Pa a hPa

  Serial.println("----- Lectura -----");
  Serial.print("Temperatura: ");
  Serial.print(temperatura);
  Serial.println(" °C");

  Serial.print("Humedad: ");
  Serial.print(humedad);
  Serial.println(" %");

  Serial.print("Presion: ");
  Serial.print(presion);
  Serial.println(" hPa");
  Serial.println("-------------------");

  delay(2000);
}
